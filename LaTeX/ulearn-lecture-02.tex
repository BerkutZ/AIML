$$
\frac{A_1,A_2,\ldots,A_n}{B}
$$

{\bf Доказать:} 
$$
A_1,A_2,\ldots,A_n\vdash B
$$
}

\uncover<+->{
{\bf Эквивалентно:} 
$$
A_1,A_2,\ldots,A_n, \neg B\vdash \square

\begin{enumerate}
\item Привести $A_1,A_2,\ldots,A_n, \neg B$ в КНФ

\end{enumerate}
} 
\end{frame}
